const initState = {
    followings: []
}
 
const followingsReducer = (state = initState, action) => {
    switch(action.type) {
        case 'addFriend':
            return {
                ...state,
                followings: [
                    ...state.followings,
                    action.friend
                ]
            }
        case 'updateFollowings':
            return {
                ...state,
                followings: action.followings
            }
        default:
            return state
    }
}
 
export default followingsReducer
